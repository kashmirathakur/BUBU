const video = document.getElementById("video"); const canvas = document.getElementById("canvas"); const captureBtn = document.getElementById("captureBtn"); const continueBtn = document.getElementById("continueBtn"); navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } }) .then(stream => { video.srcObject = stream; }); captureBtn.onclick = function () { canvas.width = video.videoWidth; canvas.height = video.videoHeight; const ctx = canvas.getContext("2d"); ctx.drawImage(video, 0, 0); // Stop the camera video.srcObject.getTracks().forEach(track => track.stop()); // Show captured image video.style.display = "none"; canvas.style.display = "block"; // Hide Capture button and show Continue captureBtn.style.display = "none"; continueBtn.style.display = "block"; }; continueBtn.onclick = function () { window.location.href = "questions.html"; };
